Download Sources

GET /api/sources/{id}

Get download links for a movie in multiple quality options (360p, 480p, 720p).

Parameters

Name Type Description
id string Movie ID (from search or info endpoints)
season string Season number(tv shows only - not applicable to single movies)ie ?season=2
episode string Episode number(tv shows only - not applicable to single movies)ie ?season=2&episode=5

Example Request(Movie)

GET https://movieapi.giftedtech.co.ke/api/sources/5099284245269335848

Example Request(Tv Show)

GET https://movieapi.giftedtech.co.ke/api/sources/9028867555875774472?season=1&episode=1

Code Examples

JavaScript

// Search for movies
async function searchMovies(query) {
  const response = await fetch(
    `https://movieapi.giftedtech.co.ke/api/search/${encodeURIComponent(query)}`
  );
  
  const data = await response.json();
  return data;
}

// Get movie info
async function getMovieInfo(movieId) {
  const response = await fetch(
    `https://movieapi.giftedtech.co.ke/api/info/${movieId}`
  );
  
  const data = await response.json();
  return data;
}

// Get download sources
async function getDownloadSources(movieId) {
  const response = await fetch(
    `https://movieapi.giftedtech.co.ke/api/sources/${movieId}`
  );
  
  const data = await response.json();
  return data;
}

// Example usage
searchMovies('Black Panther')
  .then(results => {
    console.log(results);
    const firstMovieId = results.results.items[0].subjectId;
    return getMovieInfo(firstMovieId);
  })
  .then(movieInfo => {
    console.log(movieInfo);
    return getDownloadSources(movieInfo.results.subject.subjectId);
  })
  .then(downloadSources => {
    console.log(downloadSources);
  })
  .catch(error => console.error('Error:', error));

Python

import requests
import json

# Search for movies
def search_movies(query):
    url = f"https://movieapi.giftedtech.co.ke/api/search/{query}"
    response = requests.get(url)
    return response.json()

# Get movie info
def get_movie_info(movie_id):
    url = f"https://movieapi.giftedtech.co.ke/api/info/{movie_id}"
    response = requests.get(url)
    return response.json()

# Get download sources
def get_download_sources(movie_id):
    url = f"https://movieapi.giftedtech.co.ke/api/sources/{movie_id}"
    response = requests.get(url)
    return response.json()

# Example usage
results = search_movies("Black Panther")
print(json.dumps(results, indent=2))

first_movie_id = results['results']['items'][0]['subjectId']
movie_info = get_movie_info(first_movie_id)
print(json.dumps(movie_info, indent=2))

download_sources = get_download_sources(first_movie_id)
print(json.dumps(download_sources, indent=2))
